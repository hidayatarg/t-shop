// promises
const getProducts = async(req, res, next) => {  
    pool
        .query(getAllProductsQuery)
        .then(result => {
            res.status(200).json({
                success: true,
                data: result.rows
            })
        })
        .catch(e => res.json(e.stack))

// Async await
        try {
            const result = await pool.query(getAllProductsQuery)
            res.status(200).json({
                success: true,
                data: result.rows
            })
        } catch (err) {
            res.json(err.stack)
        }
}
